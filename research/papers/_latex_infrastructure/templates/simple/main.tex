\begin{equation}

\end{equation}

% Example algorithm
\begin{algorithm}
\caption{Assembly Formation}
\label{alg:assembly_formation}
\begin{algorithmic}[1]
    \STATE \textbf{Input:} Area $\area{i}$ with $\nneurons{i}$ neurons, $\kactive$ active neurons
    \STATE \textbf{Output:} Stable assembly $\assembly{i}$
    \STATE Initialize activations $\vect{a}^{(0)} \sim \mathcal{U}(0,1)$
    \FOR{$t = 1$ to $T_{\max}$}
        \STATE Compute weighted input: $\vect{u}^{(t)} \gets \weights{} \cdot \vect{a}^{(t-1)}$
        \STATE Select top-$k$ neurons: $\vect{a}^{(t)} \gets \TopK(\vect{u}^{(t)}, \kactive)$
        \IF{$\vect{a}^{(t)} = \vect{a}^{(t-1)}$}
            \RETURN $\assembly{i} \gets \{n : a_n^{(t)} > 0\}$
        \ENDIF
    \ENDFOR
    \RETURN Failure (did not converge)
\end{algorithmic}
\end{algorithm}
